6 /* Possible inflate modes between inflate() calls */
7 typedef enum {
8  HEAD, /* i: waiting for magic header */
9  FLAGS, /* i: waiting for method and flags (gzip) */
10  TIME, /* i: waiting for modification time (gzip) */
11  OS, /* i: waiting for extra flags and operating system (gzip) */
12  EXLEN, /* i: waiting for extra length (gzip) */
13  EXTRA, /* i: waiting for extra bytes (gzip) */
14  NAME, /* i: waiting for end of file name (gzip) */
15  COMMENT, /* i: waiting for end of comment (gzip) */
16  HCRC, /* i: waiting for header crc (gzip) */
17  DICTID, /* i: waiting for dictionary check value */
18  DICT, /* waiting for inflateSetDictionary() call */
19  TYPE, /* i: waiting for type bits, including last-flag bit */
20  TYPEDO, /* i: same, but skip check to exit inflate on new block */
21  STORED, /* i: waiting for stored size (length and complement) */
22  COPY_, /* i/o: same as COPY below, but only first time in */
23  COPY, /* i/o: waiting for input or output to copy stored block */
24  TABLE, /* i: waiting for dynamic block table lengths */
25  LENLENS, /* i: waiting for code length code lengths */
26  CODELENS, /* i: waiting for length/lit and distance code lengths */
27  LEN_, /* i: same as LEN below, but only first time in */
28  LEN, /* i: waiting for length/lit/eob code */
29  LENEXT, /* i: waiting for length extra bits */
30  DIST, /* i: waiting for distance code */
31  DISTEXT, /* i: waiting for distance extra bits */
32  MATCH, /* o: waiting for output space to copy string */
33  LIT, /* o: waiting for output space to write literal */
34  CHECK, /* i: waiting for 32-bit check value */
35  LENGTH, /* i: waiting for 32-bit length (gzip) */
36  DONE, /* finished check, done -- remain here until reset */
37  BAD, /* got a data error -- remain here until reset */
38  MEM, /* got an inflate() memory error -- remain here until reset */
39  SYNC /* looking for synchronization bytes to restart inflate() */
40 } inflate_mode;

42 /* state maintained between inflate() calls. Approximately 10K bytes. */
43 struct inflate_state {
44  inflate_mode mode; /* current inflate mode */
45  int last; /* true if processing last block */
46  int wrap; /* bit 0 true for zlib, bit 1 true for gzip */
47  int havedict; /* true if dictionary provided */
48  int flags; /* gzip header method and flags (0 if zlib) */
49  unsigned dmax; /* zlib header max distance (INFLATE_STRICT) */
50  unsigned long check; /* protected copy of check value */
51  unsigned long total; /* protected copy of output count */
52  /* sliding window */
53  unsigned wbits; /* log base 2 of requested window size */
54  unsigned wsize; /* window size or zero if not using window */
55  unsigned whave; /* valid bytes in the window */
56  unsigned wnext; /* window write index */
57  unsigned char *window; /* allocated sliding window, if needed */
58  /* bit accumulator */
59  unsigned long hold; /* input bit accumulator */
60  unsigned bits; /* number of bits in "in" */
61  /* for string and stored block copying */
62  unsigned length; /* literal or length of data to copy */
63  unsigned offset; /* distance back to copy string from */
64  /* for table and code decoding */
65  unsigned extra; /* extra bits needed */
66  /* fixed and dynamic code tables */
67  code const *lencode; /* starting table for length/literal codes */
68  code const *distcode; /* starting table for distance codes */
69  unsigned lenbits; /* index bits for lencode */
70  unsigned distbits; /* index bits for distcode */
71  /* dynamic table building */
72  unsigned ncode; /* number of code length code lengths */
73  unsigned nlen; /* number of length code lengths */
74  unsigned ndist; /* number of distance code lengths */
75  unsigned have; /* number of code lengths in lens[] */
76  code *next; /* next available space in codes[] */
77  unsigned short lens[320]; /* temporary storage for code lengths */
78  unsigned short work[288]; /* work area for code table building */
79  code codes[ENOUGH]; /* space for code tables */
80  int sane; /* if false, allow invalid distance too far */
81  int back; /* bits back of last unprocessed length/lit */
82  unsigned was; /* initial length of match */
83 };
